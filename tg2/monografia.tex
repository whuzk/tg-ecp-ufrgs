% ----------------------------------------------------------
% Classe do documento
% ----------------------------------------------------------
\documentclass[
    % -- opções da classe memoir --
    12pt,               % tamanho da fonte
    openright,          % capítulos começam em pág ímpar (insere página vazia caso preciso)
    twoside,            % para impressão em verso e anverso. Oposto a oneside
    a4paper,            % tamanho do papel. 
    % -- opções da classe abntex2 --
    chapter=TITLE,      % títulos de capítulos convertidos em letras maiúsculas
    %section=TITLE,     % títulos de seções convertidos em letras maiúsculas
    %subsection=TITLE,  % títulos de subseções convertidos em letras maiúsculas
    %subsubsection=TITLE,% títulos de subsubseções convertidos em letras maiúsculas
    % -- opções do pacote babel --
    english,            % idioma adicional para hifenização
    brazil              % o último idioma é o principal do documento
    ]{abntex2}

% ----------------------------------------------------------
% Pacotes
% ----------------------------------------------------------
\usepackage{styles/mystyle}

% ----------------------------------------------------------
% Informações de capa e folha de rosto
% ----------------------------------------------------------
\input{input/topmatter}

% ----------------------------------------------------------
% Indice
% ----------------------------------------------------------
\makeindex

% ----------------------------------------------------------
% Início do documento
% ----------------------------------------------------------
\begin{document}

% ----------------------------------------------------------
% Configuraçoes extra
% ----------------------------------------------------------
%faz a contagem das figuras dentro do capítulo
\counterwithin{figure}{chapter}
%faz a contagem das tabelas dentro do capítulo	
\counterwithin{table}{chapter}
% Retira espaço extra obsoleto entre as frases
\frenchspacing 

% ----------------------------------------------------------
% Pré-texto
% ----------------------------------------------------------
\imprimircapa
\cleardoublepage

\tableofcontents*
\cleardoublepage

\input{input/prelims}
\cleardoublepage

\input{input/abbrev}
\cleardoublepage

\listoffigures
\cleardoublepage

\listoftables
\cleardoublepage

\lstlistoflistings
\cleardoublepage

% ----------------------------------------------------------
% Resumos
% ----------------------------------------------------------
\input{input/resumo}
\cleardoublepage

\input{input/abstract}
\cleardoublepage

% ----------------------------------------------------------
% Texto
% ----------------------------------------------------------
\textual
\pagestyle{simple}
\input{input/intro}
\input{input/chapter2}
\input{input/chapter3}
\input{input/chapter4}
\input{input/chapter5}
\input{input/chapter6}
\input{input/chapter7}
\input{input/chapter8}

% ----------------------------------------------------------
% Pós-texto
% ----------------------------------------------------------
\postextual
\bibliography{monografia}
%\glossary

% ----------------------------------------------------------
% Apêndices
% ----------------------------------------------------------
\begin{apendicesenv}
% Imprime uma página indicando o início dos apêndices
\partapendices
\input{input/appendixA}
\end{apendicesenv}

% ----------------------------------------------------------
% Anexos
% ----------------------------------------------------------
\input{input/attachments}

% ----------------------------------------------------------
% Índice
% ----------------------------------------------------------
\phantompart
\printindex

% ----------------------------------------------------------
% Final do documento
% ----------------------------------------------------------
\end{document}
